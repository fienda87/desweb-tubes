import { createRouter, createWebHashHistory } from "vue-router";
import Homepage from "@/pages/homepage.vue";
import Howtoplay from "@/pages/howtoplay.vue";
import Tokenomics from "@/pages/tokenomics.vue";
import Refferal from "@/pages/refferal.vue";
import Leaderboard from "@/pages/leaderboard.vue";
import Aboutme from "@/pages/aboutme.vue"; // ✅ Tambah import

const routes = [
  { path: "/", name: "homepage", component: Homepage },
  { path: "/howtoplay", name: "howtoplay", component: Howtoplay },
  { path: "/tokenomics", name: "tokenomics", component: Tokenomics },
  { path: "/refferal", name: "refferal", component: Refferal },
  { path: "/leaderboard", name: "leaderboard", component: Leaderboard },
  { path: "/about", name: "aboutme", component: Aboutme }, // ✅ Tambah route
];

const router = createRouter({
  history: createWebHashHistory(),
  routes,
  scrollBehavior(to, from, savedPosition) {
    // ✅ Scroll to top saat pindah halaman
    if (savedPosition) {
      return savedPosition;
    } else {
      return { top: 0, behavior: 'smooth' };
    }
  },
});

export default router;
